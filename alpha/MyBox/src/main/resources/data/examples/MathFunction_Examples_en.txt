function f1(x) {
    var multiply = 1;
    for (var i = 1; i <= Math.abs(x); i++) {
        multiply = multiply * i;
    }
    return multiply ;
}
function f2(x) {
    var sum = 0;
    for (var i = 1; i > x; i--) {
        sum += f1(i);
    }
    return sum;
}
