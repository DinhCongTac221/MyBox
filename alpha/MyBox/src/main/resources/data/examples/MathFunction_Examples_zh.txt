function f1(x) {
    var multiply = 1;
    for (var i = 1; i <= Math.abs(x); i++) {
        multiply = multiply * i;
    }
    return multiply ;
}
function f2(x) {
    var sum = 0;
    for (var i = 1; i > x; i--) {
        sum += f1(i);
    }
    return sum;
}
if ( x > 1 ) 
    f1(x);
else if ( x == 1)
    38; 
else 
    f2(x);

var xStd = 2; 
var yMean = 2;
var yStd = 1; 
var coefficient = 0.2; 
var dx = x - xMean;
var dy = y - yMean;
var xyStd = xStd * yStd;
var dco = 1 - coefficient * coefficient;
var px = dx * dx / ( xStd * xStd );
var py = dy * dy / ( yStd * yStd );
var pxy = 2 * coefficient * dx * dy / xyStd;
var expP = - (px + py - pxy) / ( 2 * dco)
var div = 2 * Math.PI * xyStd * Math.sqrt(dco);
Math.exp(expP) / div
